Example 2: LangGraph Integration

from langgraph.graph import StateGraph
from secnode import GuardNode, TricerSecurityState

# Define secure state
class AgentState(TricerSecurityState):
  query: str
  result: str

# Add security to workflow
guard = GuardNode.create("maximum_security")
workflow = StateGraph(AgentState)
workflow.add_node("security", guard.invoke)
workflow.set_entry_point("security")
